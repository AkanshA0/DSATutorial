Linked Lists

A singly linked list stores nodes that each reference the next node. Traversal visits nodes in order by following next references.

Example: Linked Lists across languages

javascript 
class ListNode {
  constructor(val, next=null) { this.val = val; this.next = next; }
}
function traverse(head) {
  const out = [];
  let cur = head;
  while (cur) { out.push(cur.val); cur = cur.next; }
  return out;
}

python 
class ListNode:
    def __init__(self, val, next=None):
        self.val = val
        self.next = next

def traverse(head):
    out = []
    cur = head
    while cur:
        out.append(cur.val)
        cur = cur.next
    return out

java 
class ListNode {
  int val;
  ListNode next;
  ListNode(int v) { val = v; }
}

public class LinkedListDemo {
  public static int[] traverse(ListNode head) {
    java.util.ArrayList out = new java.util.ArrayList<>();
    ListNode cur = head;
    while (cur != null) { out.add(cur.val); cur = cur.next; }
    return out.stream().mapToInt(i->i).toArray();
  }
}
